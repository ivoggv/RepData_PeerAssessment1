```r
#Create a dataset
average=data %>% 
  group_by(interval) %>% 
  summarise(steps=mean(steps,na.rm = TRUE))

navalues=left_join(data %>% filter(is.na(steps)),average,by="interval") %>% 
      mutate(steps=steps.y) %>% select(date,interval,steps)
```
